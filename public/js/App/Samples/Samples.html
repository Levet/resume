Fibonacci Sequence
Looping 
			
				function fibonacci(iterator){
					var a = 0, b = 1;
					for(var i = 0; i < iterator; i++){
					    var c = a,
					        a = b,
					        b = c + b;
					}
					return a;
				}

				fibonacci(10)

				function fizzBuzz(countTo){
					for(var i = 1; i <= countTo; i++){
						if(i % 3 == 0 && i % 5 == 0){
							console.log("FizzBuzz")
						} else if(i % 5 == 0) {
							console.log("Buzz")
						} else if(i % 3 == 0) {
							console.log("Fizz")
						} else {
							console.log(i)
						}
					}
				};

				fizzBuzz(100);
